.....create project using pipeline script........

1.	create job click on new item
2.	put project name click on pipeline project and click ok
3.	description:  THIS IS MY FIRST NGO-PROJECT pipeline DEPLOYMENT
4.	choose GitHub project paste here GitHub url
5.	Triggers: choose GitHub hook trigger for GITScm polling
6.	pipeline script
EX:

pipeline {
    agent any

    stages {
        stage('github Integration') {
            steps {
                git branch: 'main', url: 'https://github.com/sanjana7499/ngo-website.git'
                echo 'github integration successfully done'
            }
        }
    
        stage('Build & Test') {
            steps {
                sshPublisher(publishers: [sshPublisherDesc(configName: 'jenkins', transfers: [sshTransfer(cleanRemote: false, excludes: '', execCommand: 'rsync -avh "/var/lib/jenkins/workspace/NGO-FIRST DEPLOY/" root@172.31.1.41:/opt', execTimeout: 120000, flatten: false, makeEmptyDirs: false, noDefaultExcludes: false, patternSeparator: '[, ]+', remoteDirectory: '', remoteDirectorySDF: false, removePrefix: '', sourceFiles: '')], usePromotionTimestamp: false, useWorkspaceInPromotion: false, verbose: false)])
                echo 'build successfully done'
            }
        }
        stage('SonarQube') {
            steps {
                echo 'build successfully done'
            }
        }
        stage('Docker') {
            steps {
                echo 'build successfully done'
            }
        }
        stage('Kubernetes') {
            steps {
                echo 'build successfully done'
            }
        }
        stage('deploy') {
            steps {
                sshPublisher(publishers: [sshPublisherDesc(configName: 'ansible', transfers: [sshTransfer(cleanRemote: false, excludes: '', execCommand: 'ansible-playbook /sourcecode/playbook.yml', execTimeout: 120000, flatten: false, makeEmptyDirs: false, noDefaultExcludes: false, patternSeparator: '[, ]+', remoteDirectory: '', remoteDirectorySDF: false, removePrefix: '', sourceFiles: '')], usePromotionTimestamp: false, useWorkspaceInPromotion: false, verbose: false)])
                echo 'deploy project successfully done'
            }
        }
    }
}
